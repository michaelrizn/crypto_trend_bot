def determine_trend(df):
    close = df['close'].iloc[-1]
    ma20 = df['MA20'].iloc[-1]
    ma50 = df['MA50'].iloc[-1]
    ema20 = df['EMA20'].iloc[-1]
    bb_upper = df['BB_upper'].iloc[-1]
    bb_lower = df['BB_lower'].iloc[-1]

    if close > ma20 and close > ma50 and close > ema20:
        if close > bb_upper:
            return '–ø–µ—Ä–µ–∫—É–ø–ª–µ–Ω–Ω–æ—Å—Ç—å'
        return '–±—ã—á–∏–π'
    elif close < ma20 and close < ma50 and close < ema20:
        if close < bb_lower:
            return '–ø–µ—Ä–µ–ø—Ä–æ–¥–∞–Ω–Ω–æ—Å—Ç—å'
        return '–º–µ–¥–≤–µ–∂–∏–π'
    elif ma20 > ma50:
        return '–∫—Ä–∞—Ç–∫–æ—Å—Ä–æ—á–Ω—ã–π –±—ã—á–∏–π'
    elif ma20 < ma50:
        return '–∫—Ä–∞—Ç–∫–æ—Å—Ä–æ—á–Ω—ã–π –º–µ–¥–≤–µ–∂–∏–π'
    else:
        return '–±–æ–∫–æ–≤–æ–π'

def calculate_accuracy(df):
    close = df['close'].iloc[-1]
    ma20 = df['MA20'].iloc[-1]
    diff = abs(close - ma20)
    max_diff = max(df['close']) - min(df['close'])
    accuracy = int((1 - diff / max_diff) * 100)
    return accuracy

def generate_forecast(df):
    slope, intercept, r_value, p_value, std_err = stats.linregress(range(len(df['close'])), df['close'])
    forecast = 'upward' if slope > 0 else 'downward' if slope < 0 else 'neutral'
    return forecast
